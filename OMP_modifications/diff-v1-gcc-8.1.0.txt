+         * Examples are /sys/fs/cgroup/cpuset/<cg>/tasks

+

+                for (int v=min; v<=max; ++v) {
+                    (*value_in)[length++] = v;

+                        buflen *= 2;
+                        if (!(*value_in = (int*) realloc(*value_in, buflen * sizeof(**value_in))))

+            }

+        *length_in = length;

+        *length_in = 0;
+    }

+    if (fclose(fp) < 0)

+    errno = err;
+    return errno ? -1 : 0;
